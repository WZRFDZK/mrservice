import smtplib from email.mime.multipart import MIMEMultipart from email.mime.text import MIMEText # --- CONFIGURATION DES PARAMÈTRES --- VOTRE_EMAIL = "votre.email@gmail.com" VOTRE_MOT_DE_PASSE = "votre_mot_de_passe_d_application" # Utilisez un mot de passe d'application Google SERVEUR_SMTP = "smtp.gmail.com" PORT_SMTP = 587 # --- LISTE DES INVITÉS (Nom, E-mail) --- invites = [ {"nom": "Jean Dupont", "email": "jean.dupont@example.com"}, {"nom": "Alice Martin", "email": "alice.martin@example.com"}, {"nom": "Directeur Sophie", "email": "sophie.logistique@example.com"} ] # --- CORPS DE L'E-MAIL (Format HTML Professionnel) --- html_template = """

INVITATION OFFICIELLE : GALA ANNUEL & KEYNOTE

Bonjour {nom},

Nous avons le plaisir de vous convier à l'événement exclusif marquant le bilan de l'année écoulée et les perspectives d'avenir de notre organisation.

📅 Date : Samedi 15 Novembre 2026

⏰ Heure : Accueil émargement dès 20h00 | Début de la Keynote à 20h30

📍 Lieu : Siège Central (HQ), Paris

👔 Dress Code : Business Casual / Tenue de soirée

Confirmer ma présence (R.S.V.P)

Cet e-mail est automatisé. Pour toute modification de dernière minute, veuillez contacter le comité d'organisation.

""" # --- SCRIPT D'ENVOI AUTOMATIQUE --- try: # Connexion sécurisée au serveur serveur = smtplib.SMTP(SERVEUR_SMTP, PORT_SMTP) serveur.starttls() serveur.login(VOTRE_EMAIL, VOTRE_MOT_DE_PASSE) for invite in invites: # Création du message pour chaque invité msg = MIMEMultipart("alternative") msg["From"] = VOTRE_EMAIL msg["To"] = invite["email"] msg["Subject"] = f"[IMPORTANT] Invitation officielle - {invite['nom']}" # Personnalisation du texte avec le nom de l'invité corps_personnalise = html_template.format(nom=invite["nom"]) msg.attach(MIMEText(corps_personnalise, "html")) # Envoi de l'e-mail serveur.sendmail(VOTRE_EMAIL, invite["email"], msg.as_string()) print(f"✅ Invitation envoyée avec succès à : {invite['nom']} ({invite['email']})") serveur.quit() print("\n🚀 Opération terminée. Toutes les invitations ont été envoyées de manière professionnelle.") except Exception as e: print(f"❌ Une erreur est survenue lors de l'envoi : {e}")
